// Streaming chatbot answering questions about a specific claim.
// The full claim JSON is injected as system context every call.

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers":
    "authorization, x-client-info, apikey, content-type",
};

Deno.serve(async (req) => {
  if (req.method === "OPTIONS") return new Response(null, { headers: corsHeaders });

  try {
    const apiKey = Deno.env.get("LOVABLE_API_KEY");
    if (!apiKey) throw new Error("LOVABLE_API_KEY not configured");

    const { messages, claim } = await req.json();

    const systemPrompt = `You are a medical claim adjudication assistant.
Answer ONLY from the provided claim data below. Do not hallucinate or invent values.
If information is not in the claim data, clearly say so.
Cite source documents and page numbers when relevant.

CLAIM DATA (JSON):
${JSON.stringify(claim, null, 2)}`;

    const upstream = await fetch("https://ai.gateway.lovable.dev/v1/chat/completions", {
      method: "POST",
      headers: {
        Authorization: `Bearer ${apiKey}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        model: "google/gemini-2.5-flash",
        messages: [{ role: "system", content: systemPrompt }, ...messages],
        stream: true,
      }),
    });

    if (!upstream.ok) {
      if (upstream.status === 429) {
        return new Response(JSON.stringify({ error: "Rate limit. Try again shortly." }), {
          status: 429,
          headers: { ...corsHeaders, "Content-Type": "application/json" },
        });
      }
      if (upstream.status === 402) {
        return new Response(JSON.stringify({ error: "AI credits exhausted. Add credits in Settings → Workspace → Usage." }), {
          status: 402,
          headers: { ...corsHeaders, "Content-Type": "application/json" },
        });
      }
      const t = await upstream.text();
      return new Response(JSON.stringify({ error: t }), {
        status: 500,
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    return new Response(upstream.body, {
      headers: { ...corsHeaders, "Content-Type": "text/event-stream" },
    });
  } catch (e) {
    console.error("claim-chat error:", e);
    return new Response(
      JSON.stringify({ error: e instanceof Error ? e.message : "Unknown error" }),
      { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } },
    );
  }
});

// Edge function: extracts structured fields from claim document images using
// the Lovable AI Gateway (Gemini vision). Accepts an array of {file_name, url}
// where url is a publicly accessible image (PNG/JPG). Returns the documents
// array matching our claim schema.
//
// PDFs should be rasterized to images on the client before calling this.

import "https://deno.land/x/xhr@0.1.0/mod.ts";

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers":
    "authorization, x-client-info, apikey, content-type",
};

interface PageInput {
  doc_id: string;
  file_name: string;
  page: number;
  image_url: string; // public URL or data: URL
}

const SYSTEM_PROMPT = `You are a medical claim document extraction engine.
You will receive ONE image of a single page from a healthcare claim packet
(could be a discharge summary, clinical note, procedure/angioplasty report,
lab report, X-ray, CT scan, invoice, prescription, or a patient photo).

Your job is to:
1. Classify the document type.
2. Extract all available fields with provenance.
3. Detect visual cues (hospital stamp, doctor signature, implant sticker, QR/barcode).

Be conservative. If a value is not clearly present, return null with low confidence.
Never invent values. Dates must be ISO format YYYY-MM-DD when possible.`;

const EXTRACTION_TOOL = {
  type: "function",
  function: {
    name: "extract_document",
    description: "Extract structured data from one claim document page.",
    parameters: {
      type: "object",
      properties: {
        doc_type: {
          type: "string",
          enum: [
            "discharge_summary",
            "clinical_note",
            "angioplasty_report",
            "lab_report",
            "xray",
            "ct_scan",
            "invoice",
            "prescription",
            "patient_photo",
            "unknown",
          ],
        },
        confidence: { type: "number" },
        visual_cues: {
          type: "object",
          properties: {
            hospital_stamp_present: { type: "boolean" },
            doctor_signature_present: { type: "boolean" },
            implant_sticker_present: { type: "boolean" },
            implant_sticker_keywords: {
              type: "array",
              items: { type: "string" },
            },
            qr_barcode_present: { type: "boolean" },
            qr_barcode_data: { type: "string" },
          },
          required: [
            "hospital_stamp_present",
            "doctor_signature_present",
            "implant_sticker_present",
            "implant_sticker_keywords",
            "qr_barcode_present",
            "qr_barcode_data",
          ],
        },
        fields: {
          type: "object",
          properties: {
            patient_name: { type: "object", properties: { value: { type: "string" }, confidence: { type: "number" }, text_span: { type: "string" } }, required: ["value", "confidence", "text_span"] },
            diagnosis: { type: "object", properties: { value: { type: "string" }, confidence: { type: "number" }, text_span: { type: "string" } }, required: ["value", "confidence", "text_span"] },
            procedure: { type: "object", properties: { value: { type: "string" }, confidence: { type: "number" }, text_span: { type: "string" } }, required: ["value", "confidence", "text_span"] },
            admission_date: { type: "object", properties: { value: { type: "string" }, confidence: { type: "number" }, text_span: { type: "string" } }, required: ["value", "confidence", "text_span"] },
            procedure_date: { type: "object", properties: { value: { type: "string" }, confidence: { type: "number" }, text_span: { type: "string" } }, required: ["value", "confidence", "text_span"] },
            discharge_date: { type: "object", properties: { value: { type: "string" }, confidence: { type: "number" }, text_span: { type: "string" } }, required: ["value", "confidence", "text_span"] },
            investigation_date: { type: "object", properties: { value: { type: "string" }, confidence: { type: "number" }, text_span: { type: "string" } }, required: ["value", "confidence", "text_span"] },
            total_amount: { type: "object", properties: { value: { type: "string" }, confidence: { type: "number" }, text_span: { type: "string" } }, required: ["value", "confidence", "text_span"] },
          },
        },
      },
      required: ["doc_type", "confidence", "visual_cues", "fields"],
    },
  },
};

async function extractOne(page: PageInput, apiKey: string) {
  const resp = await fetch("https://ai.gateway.lovable.dev/v1/chat/completions", {
    method: "POST",
    headers: {
      Authorization: `Bearer ${apiKey}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      model: "google/gemini-2.5-flash",
      messages: [
        { role: "system", content: SYSTEM_PROMPT },
        {
          role: "user",
          content: [
            { type: "text", text: `Extract data from this page (file: ${page.file_name}, page ${page.page}).` },
            { type: "image_url", image_url: { url: page.image_url } },
          ],
        },
      ],
      tools: [EXTRACTION_TOOL],
      tool_choice: { type: "function", function: { name: "extract_document" } },
    }),
  });

  if (!resp.ok) {
    const text = await resp.text();
    throw new Error(`AI gateway ${resp.status}: ${text}`);
  }

  const json = await resp.json();
  const call = json.choices?.[0]?.message?.tool_calls?.[0];
  if (!call) throw new Error("No tool call in response");
  const args = JSON.parse(call.function.arguments);

  return {
    doc_id: page.doc_id,
    file_name: page.file_name,
    page: page.page,
    doc_type: args.doc_type,
    confidence: args.confidence,
    visual_cues: {
      hospital_stamp: { present: args.visual_cues.hospital_stamp_present, bbox: [] },
      doctor_signature: { present: args.visual_cues.doctor_signature_present, bbox: [] },
      implant_sticker: {
        present: args.visual_cues.implant_sticker_present,
        keywords: args.visual_cues.implant_sticker_keywords ?? [],
      },
      qr_barcode: {
        present: args.visual_cues.qr_barcode_present,
        data: args.visual_cues.qr_barcode_data ?? "",
      },
    },
    extracted_fields: args.fields ?? {},
    issues: [],
  };
}

Deno.serve(async (req) => {
  if (req.method === "OPTIONS") return new Response(null, { headers: corsHeaders });

  try {
    const apiKey = Deno.env.get("LOVABLE_API_KEY");
    if (!apiKey) throw new Error("LOVABLE_API_KEY not configured");

    const { pages } = (await req.json()) as { pages: PageInput[] };
    if (!Array.isArray(pages) || pages.length === 0) {
      return new Response(JSON.stringify({ error: "pages[] required" }), {
        status: 400,
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    // Process pages sequentially to avoid rate limits
    const documents = [];
    for (const page of pages) {
      try {
        const doc = await extractOne(page, apiKey);
        documents.push(doc);
      } catch (e) {
        console.error("extract error for", page.file_name, e);
        documents.push({
          doc_id: page.doc_id,
          file_name: page.file_name,
          page: page.page,
          doc_type: "unknown",
          confidence: 0,
          visual_cues: {
            hospital_stamp: { present: false, bbox: [] },
            doctor_signature: { present: false, bbox: [] },
            implant_sticker: { present: false, keywords: [] },
            qr_barcode: { present: false, data: "" },
          },
          extracted_fields: {},
          issues: [`extraction_failed: ${e instanceof Error ? e.message : String(e)}`],
        });
      }
    }

    return new Response(JSON.stringify({ documents }), {
      headers: { ...corsHeaders, "Content-Type": "application/json" },
    });
  } catch (e) {
    console.error("extract-claim error:", e);
    const msg = e instanceof Error ? e.message : "Unknown error";
    const status = msg.includes("429") ? 429 : msg.includes("402") ? 402 : 500;
    return new Response(JSON.stringify({ error: msg }), {
      status,
      headers: { ...corsHeaders, "Content-Type": "application/json" },
    });
  }
});
